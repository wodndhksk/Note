$("body").delegate('#realMaxWt', 'focusout', function(e){
			if($(this).val() < 0){
				$(this).val(0);
				alert("음수 입력 불가");

				// $(this).focus();
			}

			var checkStyle = /[0-9]/;
			if(!checkStyle.test($(this).val())){
				$(this).val(0);
				alert(`숫자만 입력 가능합니다.`);
			}
			
			$(this).focus();
		});
