
```xml
SELECT
			 CT.CTGR_NO
			,CT.USE_SCOPE_CD
			,max(IF(CTNM.LNG_CD = 'KO', CTNM.CTGR_NM , NULL)) AS CTGR_NM_KO
			,max(IF(CTNM.LNG_CD = 'EN', CTNM.CTGR_NM , NULL)) AS CTGR_NM_EN

			,CT.SORT_SEQ
			,CT.USE_YN
			,CT.DESCRIPTION
			,CT.SERV_TYPE

		FROM STD_CTGR CT

		INNER JOIN STD_CTGR_NM CTNM

		ON CT.CTGR_NO = CTNM.CTGR_NO
		
		GROUP BY CT.CTGR_NO
```
